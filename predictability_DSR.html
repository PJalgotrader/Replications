This is the R-code for my working paper : Household financial obligations ratio and the predictability of market returns

First you need to download the .csv data files here: Annual or Quarterly

I will use annual data here to replicate the tables and graphs used in my paper. You could easily follow the same steps and check the results using quarterly data.

You need the following libraries in R to use my code: dplyr, sandwich, lmtest,tseries, ggplot2, forecast

Data_source  <- "./Predictability_annual.csv"                                    
A_data       <- read.table(Data_source, sep = ",", header = TRUE) 
A_df         <- tbl_df(A_data)                                                              #A_df  = annual data frame

Selecting the features for our replication.

H            <- 1     # this is the horizon number i.e if H=5 then we are working with 5-year returns
start_date   <- 1980
end_date     <- 2015
LHS          <- 1    # LHS variable is eithr 1= market returns , 2= excess returns , 3= CRSP_VW returns
OOS_period   <- 20   # out of sample periods for annual data 
A            <- as.data.frame(filter(A_df, year>=start_date & year<=end_date)) # can pickup any data frame by : as.data.frame
T            <- nrow(A)  # Number of time periods

if (LHS==1){ LHS_tag <- "SP500"} else  { if (LHS==2){LHS_tag<- "ERP"} else {LHS_tag<- "CRSP_VW"}}

# lets decide which predictors to compare : 

predictors     <- c("dsr_demean", "f_demean", "dp","dy", "ep",  "eqis", "ik",  "MKVGDP", "cay_ll" )
multi_tag      <- c("dsr_demean",  "dp", "ep",  "eqis", "ik",  "MKVGDP", "cay_ll" ) # this must be at leat 2 variables and is used for multiple regressions

reg_single_OLS <- list()                                    # this is where we store single   regression results (OLS) 
reg_single_NW  <- list()                                    # this is where we store single   regression results (NW corrected ones) 
reg_multi_OLS  <- list()                                    # this is where we store multiple regression results (OLS) 
reg_multi_NW   <- list()                                    # this is where we store multiple regression results (NW corrected ones) 
plot_list      <- list()                                    # this is where we store the plots (H-horizon returns vs predictors)
np             <- 0                                             # this is the number of plot


b              <- data.frame(stat=rep("b", H))                  # this is the estimate for predictor coef
intercept      <- data.frame(stat=rep("intercept", H))          # this is the estimate for intercept          
adjR2          <- data.frame(stat=rep("adj.R2", H))             # this is used for comparing adjR2
R2             <- data.frame(stat=rep("R2", H))
adjR2_multi    <- data.frame(stat=rep("adj.R2", H))             # this is used for comparing adjR2
R2_multi       <- data.frame(stat=rep("R2", H))

t_NW_b     <- data.frame(stat=rep("t_NW_b", H))             # this is the t_NW (Newywest adj) for the predictor estimate
t_NW_int   <- data.frame(stat=rep("t_NW_intercept", H))     # this is the t_NW (Newywest adj) for the intercept
p_NW_b     <- data.frame(stat=rep("p_NW_b", H))             # this is the p-value (Newywest adj) for the predictor estimate
p_NW_int   <- data.frame(stat=rep("p_NW_intercept", H))     # this is the p-value (Newywest adj) for the intercept



# out of sample variables
OOS_est_period <- OOS_period

IS_error_N     <- data.frame(stat=rep("IS_e_N", T-H+1), year=(start_date):(end_date-H+1))  
IS_error_A     <- data.frame(stat=rep("IS_e_A", T-H), year=(start_date+1):(end_date-H+1)) # because we are regressing on 1-lag
OOS_error_N    <- data.frame(stat=rep("OOS_e_N", T-H+2- OOS_est_period), year=(start_date+ OOS_est_period-1) :(end_date-H+1))
OOS_error_A    <- data.frame(stat=rep("OOS_e_A", T-H+2- OOS_est_period), year=(start_date+ OOS_est_period-1) :(end_date-H+1)) 
OOS_error_AR   <- data.frame(stat=rep("OOS_e_AR",T-H+2- OOS_est_period), year=(start_date+ OOS_est_period-1) :(end_date-H+1)) 
MSE_N          <- data.frame(stat=rep("MSE_N",1))  # this is the mean squared error for OOS test. N= Null hypothesis = historical mean model
MSE_A          <- data.frame(stat=rep("MSE_A",1))  # this is the mean squared error for OOS test. A= Alternative hypothesis = conditional model
MSE_AR         <- data.frame(stat=rep("MSE_AR",1))  # this is the mean squared error for OOS test. AR= Autoregressive benchmark (ll2001)        
IS_R2          <- data.frame(stat=rep("IS_R2",1))  # this is In sample R2 for H-year forecastes
IS_adj.R2      <- data.frame(stat=rep("IS_adj.R2",1))  # this is In sample adj.R2 for H-year forecastes

OOS_R2_N       <- data.frame(stat=rep("OOS_R2_N",1))  # this is out of sample R2 for H-year forecastes when the benchmark is historical mean
OOS_R2_AR      <- data.frame(stat=rep("OOS_R2_AR",1))  # this is out of sample R2 for H-year forecastes when the benchmark is autoregressive

delta_RMSE_N   <- data.frame(stat=rep("delta_RMSE_N",1))  # this is delta_RMSE when the benchmark is historical mean 
delta_RMSE_AR  <- data.frame(stat=rep("delta_RMSE_AR",1))  # this is delta_RMSE when the benchmark is autoregressive

MSE_F_N        <- data.frame(stat=rep("MSE_F_N",1))       # this is MSE_F when the benchmark is historical mean 
MSE_F_AR       <- data.frame(stat=rep("MSE_F_AR",1))       # this is MSE_F when the benchmark is autoregressive



IS_CSSE        <- data.frame(year=(start_date+1) :(end_date-H+1), stat=rep("IS_CSSE",T-H))       # this is IS cumulative SSE (loosing 1 observation)
OOS_CSSE       <- data.frame(year=(start_date+ OOS_est_period-1) :(end_date-H+1), stat=rep("OOS_CSSE",T-H+2- OOS_est_period ))       # this is OOS cumulative SSE

plot_OOS_list  <- list()                 # this is where we store the OOS plots 

# the follwing 3 are not used in this paper
delta_MAE      <- data.frame(stat=rep("delta_MAE",1))   # this is delta_MAE  
MSE_T          <- data.frame(stat=rep("MSE_T",1))       # this is MSE_T
ENC            <- data.frame(stat=rep("ENC",1))       # this is ENC
#################################################################################


for(k in predictors){
        
        RHS          <- k   # this is the right hand side variable : to see list of variables type in : names(A)
        
        # making statistics data frame for horizons in 1: H
        reg_df              <- data.frame(Intercept=numeric(H), b=numeric(H), 
                                          R2=numeric(H), adj.R2=numeric(H),
                                          t_ols_int = numeric(H), t_ols_b   = numeric(H) ,
                                          t_NW_int  = numeric(H), t_NW_b    = numeric(H), 
                                          p_ols_int = numeric(H), p_ols_b    = numeric(H),
                                          p_NW_int  = numeric(H), p_NW_b     = numeric(H))
        
        
        
        # regressions of mean returns on predicting factor
        
        for(j in 1:H){
                
                # making mean returns 
                sp_H    <- c()        # the SP500 return
                ERP_H   <- c()        # the excess return
                VW_H    <- c()        # the CRSP_VW return
                
                for (i in 1:(T-j+1)){
                        sp_H[i]     <-  mean(A$SP500 [i:(i+j-1)])
                        ERP_H[i]    <-  mean(A$ERP [i:(i+j-1)])
                        VW_H[i]     <-  mean(A$CRSP_SPvw [i:(i+j-1)])
                }
                
                # picking the left hand side variable
                if (LHS==1){ LHS_ret<- sp_H} else  { if (LHS==2){LHS_ret<- ERP_H} else {LHS_ret<-VW_H}}
                
                # regressions and statistics
                reg_H                  <- lm(LHS_ret~lag(A[1:(T-j+1),RHS],1)) 
                
                #storing single regressions results for H horizons
                reg_single_OLS[[j]]    <- reg_H
                reg_single_NW[[j]]     <- coeftest(reg_H, vcov = NeweyWest)        
                
                reg_df[j,"Intercept"]  <- coef(reg_H)[1]
                reg_df[j,"b"]          <- coef(reg_H)[2]
                reg_df[j,"R2"]         <- summary(reg_H)$r.squared
                reg_df[j,"adj.R2"]     <- summary(reg_H)$adj.r.squared
                reg_df[j,"t_ols_int"]  <- coef(summary(reg_H))[1,3]
                reg_df[j,"t_ols_b"]    <- coef(summary(reg_H))[2,3]
                reg_df[j,"p_ols_int"]  <- coef(summary(reg_H))[1,4]
                reg_df[j,"p_ols_b"]    <- coef(summary(reg_H))[2,4]
                reg_df[j,"t_NW_int"]   <- coeftest(reg_H, vcov=NeweyWest)[1,3]
                reg_df[j,"t_NW_b"]     <- coeftest(reg_H, vcov=NeweyWest)[2,3]
                reg_df[j,"p_NW_int"]   <- coeftest(reg_H, vcov=NeweyWest)[1,4]
                reg_df[j,"p_NW_b"]     <- coeftest(reg_H, vcov=NeweyWest)[2,4]
                
                
                # multiple regressions , pick what predictors you want to include here
                
                reg_multi              <- lm(LHS_ret ~ sapply(A[1:(T-j+1), multi_tag], lag)) 
                reg_multi_OLS[[j]]     <- reg_multi
                R2_multi[j,"R2"]       <- summary(reg_multi)$r.squared
                adjR2_multi[j,"adj.R2"]<- summary(reg_multi)$adj.r.squared 
                reg_multi_NW[[j]]      <- coeftest(reg_multi, vcov = NeweyWest)
        }
        
        
        
        # Generating out of sample tests for H-year horizon
        
        #In Sample (IS) 
        #1. Historical mean model
        
        IS_LHS          <- as.matrix(LHS_ret[1:(T-H+1)])
        IS_error_N[,k]  <- IS_LHS - mean(IS_LHS, na.rm=TRUE)  
        
        
        #2. OLS model
        reg            <- lm(LHS_ret~lag(A[1:(T-H+1),RHS],1))
        IS_error_A[,k] <- reg$residuals
        IS_R2[,k]      <- summary(reg)$adj.r.squared
        IS_adj.R2[,k]  <- summary(reg)$adj.r.squared
        
        
        # OUT-of-Sample (OOS) : Only use information that is available up to the time at which the forecast is made
        
        temp_e_N      <- c()   # temporary errors of historical mean benchmark model
        temp_e_A      <- c()   # temporary errors of  Alternative (conditional) model
        temp_e_AR     <- c()   # temporary errors of Auto regressive benchmark model
        
        for (s in (1 + OOS_est_period ):(T-H+1)) {     
                
                #1. Historical mean model , with (s-1) observation we estimate the numbers for _[s]
                OOS_LHS            <- as.matrix(LHS_ret)
                temp_e_N[s]        <- OOS_LHS[s] - mean(OOS_LHS[1:(s-1)])  
                
                #2. AR benchmart 
                reg_OOS_AR         <- lm(OOS_LHS[1:(s-1)]~lag(OOS_LHS[1:(s-1)],1))
                temp_e_AR[s]       <- (coef(reg_OOS_AR)[1]+coef(reg_OOS_AR)[2]*A[(s-1),RHS])  - OOS_LHS[s]
                
                
                #3. OLS model
                reg_OOS            <- lm(OOS_LHS[1:(s-1)]~lag(A[1:(s-1),RHS],1))
                temp_e_A[s]        <-  (coef(reg_OOS)[1]+coef(reg_OOS)[2]*A[(s-1),RHS])  - OOS_LHS[s]
        }
        
        OOS_error_N[,k]    <- as.matrix( c(0, temp_e_N[(1 + OOS_est_period):(T-H+1)]))
        OOS_error_A[,k]    <- as.matrix( c(0, temp_e_A[(1 + OOS_est_period):(T-H+1)]))
        OOS_error_AR[,k]   <- as.matrix( c(0, temp_e_AR[(1 + OOS_est_period):(T-H+1)]))
        
        
        # constructing Goyal and Welch OOS tests
        
        T_OOS      <- T-H+1- OOS_est_period             # T is the total number of forecast observations
        h          <- H                                # h is the overlap degree
        
        MSE_N[,k]  <- mean(OOS_error_N[,k]^2)
        MSE_A[,k]  <- mean(OOS_error_A[,k]^2)
        MSE_AR[,k] <- mean(OOS_error_AR[,k]^2)
        
        
        OOS_R2_N[,k]     <- 1 - MSE_A[,k]/MSE_N[,k]
        OOS_R2_AR[,k]    <- 1 - MSE_A[,k]/MSE_AR[,k]
        
        delta_RMSE_N[,k]  <- sqrt(MSE_N[,k]) - sqrt(MSE_A[,k])
        delta_RMSE_AR[,k] <- sqrt(MSE_AR[,k]) - sqrt(MSE_A[,k])
        
        MSE_F_N[,k]       <- (T_OOS - h +1) * ((MSE_N[,k]-MSE_A[,k])/MSE_A[,k])
        MSE_F_AR[,k]      <- (T_OOS - h +1) * ((MSE_AR[,k]-MSE_A[,k])/MSE_A[,k])
        
        
        
        # not used in this paper again!
        delta_MAE[,k]  <- sum((abs(OOS_error_N[,k])-abs(OOS_error_A[,k])))/T_OOS  # loosing 1 observation for _A
        d              <-  OOS_error_N[,k] - OOS_error_A[,k]       
        MSE_T[,k]      <- sqrt(T_OOS + 1 - 2* h + h*(h-1)/T_OOS) * (mean(d)/sqrt(var(d)))
        ENC[,k]        <- ((T_OOS-h+1)/T_OOS) * sum(OOS_error_N[,k]^2 - OOS_error_N[,k]*OOS_error_A[,k] ) / MSE_A[,k]
        #################################
        
        # Cumulative SSE difference
        
        IS_CSSE[,k]  <- cumsum(IS_error_N[-1,k]^2)-cumsum(IS_error_A[,k]^2) # we lose 1 observation in IS_A regression
        OOS_CSSE[,k] <- cumsum(OOS_error_N[,k]^2)-cumsum(OOS_error_A[,k]^2)
        
        df1_IS   <- data.frame(x=(start_date+1) :(end_date-H+1), y=IS_CSSE[,k] , df_name= "IS" )
        df1_IS$y <- df1_IS$y - df1_IS$y[OOS_est_period-1] 
        #Because we lose one observation due to the lag we Shift IS errors vertically, so that the IS line begins 
        # at zero on the date of first OOS prediction. (Goyal/Welch (2008, p. 1465))
        
        
        df2_OOS   <- data.frame(x=(start_date+ OOS_est_period-1) :(end_date-H+1),y=OOS_CSSE[,k], df_name="OOS" )
        df_OOS    <- rbind(df1_IS,df2_OOS)
        
        
        # Plotting OOS vs IS cumulative SSE
        p_OOS    <-   ggplot(df_OOS, aes(y=y, x=x, linetype= df_name)) + geom_line(size=1)+
                labs(x = paste("YEAR ", "( OOS Estimation Period =",OOS_est_period ,", H=",H,  ")", sep=" " ),y="Cumulative SSE Difference", linetype=paste("(",LHS_tag,",", k,")", sep=" " ))+
                geom_hline(aes(yintercept=0))+
                coord_cartesian( ylim = c(-0.2,0.2)) + theme_bw()
        
        
        
        
        # Plotting H-Horizon returns vs predictors
        if (k=="dsr_demean" | k=="f_demean" | k=="MKVGDP" | k=="ik") {teta<- -1} else {teta<- 1}    
        # teta is used to flip the graphs for predictors with negative coef
        
        df1   <- data.frame( y= teta*(A[,RHS]  - mean(A[,RHS]))/sqrt(var(A[,RHS])),     x= start_date:end_date , var_name= RHS)
        df2   <- data.frame( y= (LHS_ret -mean(LHS_ret))/sqrt(var(LHS_ret)) , x= start_date:(start_date+T-H) , var_name= LHS_tag)
        df    <- rbind(df1, df2)
        
        p   <-   ggplot(df, aes(y=y, x=x, linetype= var_name)) + geom_line(size=1)+
                labs(x = paste("YEAR ", "( H:",H, ", R2:",round(100*IS_R2[,k],2), "% )", sep=" " ),y="Standardize units", linetype=NULL) + theme_bw()
        # ggtitle("Forecast and actual 5-year returns")  +
        # coord_cartesian(xlim =  , ylim = , expand = TRUE)+
        #coord_fixed(ratio=130)+  #  ratio aspect ratio, expressed as y / x
        #theme(text = element_text(size=16)) + #size of axis lables
        #theme(axis.text = element_text( color = "black", size = 12))
        
        # storing the plots
        np                   <- np + 1
        plot_OOS_list[[np]]  <- p_OOS
        plot_list[[np]]      <- p
        
        b[,k]            <- reg_df[,"b"]
        intercept[,k]    <- reg_df[,"Intercept"]
        R2[,k]           <- reg_df[,"R2"]
        adjR2[,k]        <- reg_df[,"adj.R2"]
        t_NW_b[,k]       <- reg_df[,"t_NW_b"]
        t_NW_int[,k]     <- reg_df[,"t_NW_int"]
        p_NW_b[,k]       <- reg_df[,"p_NW_b"]
        p_NW_int[,k]     <- reg_df[,"p_NW_int"]
        
}

Now let’s look at some out of sample graphs.

Out-Of-Sample performance of different conditional models

These figures plot the IS and OOS performance of different conditional models. The lines are the cumulative squared prediction errors of the benchmark model (historical mean) minus the cumulative squared prediction error of the conditional model. The conditional model is a model that relies on predictive variables noted in each graph. The IS prediction relative performance is the red line and the OOS prediction relative perfomance is the blue line. An increase in a line indicates better performance of the named model; a decrease in a line indicates better performance of the historical mean model.
